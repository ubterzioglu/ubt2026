## 8. Appointment Booking System (NEW)

### 8.1 Database Schema (Supabase)

#### Table: `appointment_slots`
```sql
CREATE TABLE appointment_slots (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  start_at TIMESTAMPTZ NOT NULL,
  duration_minutes INTEGER NOT NULL CHECK (duration_minutes IN (15, 30)),
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_appointment_slots_start_at ON appointment_slots(start_at);
CREATE INDEX idx_appointment_slots_active ON appointment_slots(is_active);
```

#### Table: `appointments`
```sql
CREATE TABLE appointments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  slot_id UUID NOT NULL REFERENCES appointment_slots(id) ON DELETE CASCADE,
  full_name TEXT NOT NULL,
  whatsapp_phone TEXT NOT NULL,
  contact_key TEXT NOT NULL,
  topic TEXT NOT NULL,
  linkedin_url TEXT,
  description TEXT,
  status TEXT DEFAULT 'new' CHECK (status IN ('new', 'approved', 'completed', 'cancelled')),
  admin_note TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(slot_id)
);

CREATE INDEX idx_appointments_contact_key ON appointments(contact_key);
CREATE INDEX idx_appointments_status ON appointments(status);
```
